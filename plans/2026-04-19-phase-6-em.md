# Phase 6 EM Implementation Plan

> **For agentic workers:** Execute task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the `em_iterations` opt-in loop around the threshold-joint baseline so that discrete-trait (binary + ordinal + OVR categorical) imputation improves on correlated-discrete datasets.

**Architecture:** One new argument (`em_iterations = 0L`) on `fit_baseline()` and `impute()`. When >= 1, a new `fit_joint_threshold_baseline_em()` wraps the existing single-pass `fit_joint_threshold_baseline()` and feeds `Σ` back as diagonal per-trait prior SD between iterations. OVR categorical iterates per-trait with a length-K variance vector `σ_cat`. Default behaviour unchanged at `em_iterations = 0L` (byte-identical to v0.9.1).

**Tech Stack:** R 4.x, `testthat` 3, `roxygen2`, `devtools`, `Rphylopars`, `pigauto` 0.9.1.

  categorical) iterates: the BM rate `Σ` learned by `Rphylopars::phylopars()`
  at iteration `k` is fed back as the per-trait prior SD at iteration

  discrete-trait accuracy typically gains a few pp.
- Full off-diagonal conditioning on `Σ` is deferred to a future Phase 7.
  Phase 6 uses `sqrt(diag(Σ))` only.

```r
# Extract per-column BM rate variance from an Rphylopars fit's $sigma2.
# Returns a named numeric vector aligned to liab_cols that were actually fit
# (i.e. corresponds to `fit_cols` in fit_joint_threshold_baseline).
# Phase 6 EM uses this to build next iteration's sd_prior_vec.
#' @keywords internal
#' @noRd
extract_liability_variances <- function(phylopars_fit) {
  sig2 <- phylopars_fit$pars$sigma2
  if (is.matrix(sig2)) {
    as.numeric(diag(sig2))
  } else {
    as.numeric(sig2)
  }
}

# Relative Frobenius distance between two variance vectors (or matrices).
# Used as the Phase 6 EM early-stop criterion: delta = ||v_new - v_old||_F
# / ||v_old||_F.  Returns +Inf if v_old has zero norm (first real iter).
#' @keywords internal
#' @noRd
rel_frobenius <- function(v_new, v_old) {
  den <- sqrt(sum(v_old^2))
  if (!is.finite(den) || den == 0) return(Inf)
  num <- sqrt(sum((v_new - v_old)^2))
  num / den
}
```

```r
# Phase 6 EM wrapper around fit_joint_threshold_baseline().
# Iterates:
#   1. Build liability matrix with per-trait sd_prior_vec (= sqrt(diag(Σ))).
#   2. Run phylopars to get new Σ.
#   3. Check relative Frobenius convergence.
# Returns the final single-pass baseline result augmented with $em_state.
#' @keywords internal
#' @noRd
fit_joint_threshold_baseline_em <- function(data, tree, splits,
                                             graph = NULL,
                                             soft_aggregate = FALSE,
                                             em_iterations = 5L,
                                             em_tol = 1e-3) {
  stopifnot(joint_mvn_available(), em_iterations >= 1L)

  # Discover the number of liability columns on iter 1 (plug-in path).
  # Start sd_prior_vec as NULL so build_liability_matrix uses sd=1.
  sd_prior_vec <- NULL
  prev_vars    <- NULL
  delta        <- Inf
  base         <- NULL
  iter_run     <- 0L

  for (iter in seq_len(em_iterations)) {
    built <- build_liability_matrix(data, splits = splits,
                                     soft_aggregate = soft_aggregate,
                                     sd_prior_vec = sd_prior_vec)

    # Delegate to the single-pass fit, but pass the already-built liability
    # matrix to avoid re-building it.
    # We inline the phylopars call here because fit_joint_threshold_baseline()
    # internally calls build_liability_matrix and would overwrite our sd_prior.
    new_base <- .phylopars_on_built(built, data, tree)

    if (is.null(new_base)) {
      # phylopars failed. Fall back to the previous iteration's base if any.
      if (is.null(base)) {
        stop("fit_joint_threshold_baseline_em: phylopars failed on iter 1; ",
             "no fallback available. Try em_iterations = 0L.", call. = FALSE)
      }
      warning("fit_joint_threshold_baseline_em: phylopars failed at iter ",
              iter, ". Returning previous iteration's baseline.", call. = FALSE)
      break
    }

    iter_run <- iter
    base     <- new_base

    # Compute next-iter sd_prior from this iter's Σ diagonal.
    new_vars <- extract_liability_variances(base$phylopars_fit)

    if (!is.null(prev_vars)) {
      delta <- rel_frobenius(new_vars, prev_vars)
      if (delta < em_tol) break
    }

    # prev_vars is NULL on iter 1. Otherwise it's iter-k's variances.
    prev_vars    <- new_vars
    # Build sd_prior_vec aligned to the full liab_cols (may be longer than
    # new_vars if some cols were skipped by the <2-obs guard). Fill unfitted
    # cols with 1 (plug-in fallback).
    sd_prior_vec <- rep(1, length(built$liab_cols))
    sd_prior_vec[base$fit_cols_idx] <- sqrt(pmax(new_vars, 1e-8))
  }

  base$em_state <- list(
    iterations_run = iter_run,
    converged      = is.finite(delta) && delta < em_tol,
    final_delta    = delta
  )
  base
}

# Internal: run phylopars on an already-built liability matrix. Returns a
# list shaped like fit_joint_threshold_baseline's output, plus $phylopars_fit
# and $fit_cols_idx. NULL on phylopars failure.
.phylopars_on_built <- function(built, data, tree) {
  X_liab     <- built$X_liab
  liab_cols  <- built$liab_cols
  liab_types <- built$liab_types
  spp <- if (!is.null(data$species_names)) data$species_names else rownames(data$X_scaled)

  has_obs  <- apply(X_liab, 2, function(col) sum(!is.na(col)) >= 2L)
  fit_cols <- which(has_obs)
  n_species  <- nrow(X_liab)
  n_all_cols <- ncol(X_liab)
  mu_liab    <- matrix(NA_real_, nrow = n_species, ncol = n_all_cols,
                        dimnames = list(spp, colnames(X_liab)))
  se_liab    <- matrix(NA_real_, nrow = n_species, ncol = n_all_cols,
                        dimnames = list(spp, colnames(X_liab)))

  if (length(fit_cols) < 1L) {
    return(list(mu_liab = mu_liab, se_liab = se_liab,
                liab_cols = liab_cols, liab_types = liab_types,
                phylopars_fit = NULL, fit_cols_idx = integer(0)))
  }

  X_fit <- X_liab[, fit_cols, drop = FALSE]
  orig_colnames   <- colnames(X_fit)
  safe_colnames   <- make.names(orig_colnames, unique = TRUE)
  colnames(X_fit) <- safe_colnames
  df_in          <- as.data.frame(X_fit)
  df_in$species  <- spp
  df_in          <- df_in[, c("species", safe_colnames), drop = FALSE]

  fit <- tryCatch(
    Rphylopars::phylopars(trait_data = df_in, tree = tree, model = "BM"),
    error = function(e) NULL
  )
  if (is.null(fit)) return(NULL)

  tip_rows <- match(spp, rownames(fit$anc_recon))
  mu_fit   <- fit$anc_recon[tip_rows, , drop = FALSE]
  se_fit   <- sqrt(fit$anc_var[tip_rows, , drop = FALSE])
  colnames(mu_fit) <- orig_colnames
  colnames(se_fit) <- orig_colnames
  mu_liab[, fit_cols] <- mu_fit
  se_liab[, fit_cols] <- se_fit

  list(mu_liab    = mu_liab,
       se_liab    = se_liab,
       liab_cols  = liab_cols,
       liab_types = liab_types,
       phylopars_fit = fit,
       fit_cols_idx  = fit_cols)
}
```

```r
# Phase 6 EM wrapper around fit_ovr_categorical_fits().
# Iterates a length-K variance vector `sigma_cat` per-categorical-trait;
# on each iter, runs K OVR fits, each with sd_prior = sqrt(sigma_cat[k]).
# Returns the same shape as fit_ovr_categorical_fits() plus $em_state_per_trait.
#' @keywords internal
#' @noRd
fit_ovr_categorical_fits_em <- function(data, tree, trait_name,
                                         splits = NULL, graph = NULL,
                                         em_iterations = 5L, em_tol = 1e-3) {
  stopifnot(em_iterations >= 1L)

  # Discover K by running one zero-iter call (cheap).
  tm <- data$trait_map[[trait_name]]
  K  <- length(tm$levels)

  sigma_cat  <- rep(1, K)
  prev_sigma <- NULL
  delta      <- Inf
  per_k_fits <- NULL
  iter_run   <- 0L

  for (iter in seq_len(em_iterations)) {
    # Run K OVR fits with current sigma_cat.
    iter_fits <- fit_ovr_categorical_fits(
      data        = data,
      tree        = tree,
      trait_name  = trait_name,
      splits      = splits,
      graph       = graph,
      sd_prior_k  = sqrt(pmax(sigma_cat, 1e-8))
    )
    if (is.null(iter_fits)) {
      warning("fit_ovr_categorical_fits_em: OVR fits returned NULL at iter ",
              iter, "; returning previous iter's fits.", call. = FALSE)
      break
    }
    iter_run   <- iter
    per_k_fits <- iter_fits

    new_sigma <- vapply(iter_fits$per_class_fits, function(cf) {
      if (is.null(cf) || is.null(cf$phylopars_fit)) return(NA_real_)
      vv <- extract_liability_variances(cf$phylopars_fit)
      if (length(vv) == 0L) NA_real_ else vv[1]
    }, numeric(1))
    new_sigma[is.na(new_sigma)] <- sigma_cat[is.na(new_sigma)]  # keep prior on failure

    if (!is.null(prev_sigma)) {
      delta <- rel_frobenius(new_sigma, prev_sigma)
      if (delta < em_tol) break
    }
    prev_sigma <- new_sigma
    sigma_cat  <- new_sigma
  }

  per_k_fits$em_state <- list(
    iterations_run = iter_run,
    converged      = is.finite(delta) && delta < em_tol,
    final_delta    = delta,
    sigma_cat      = sigma_cat
  )
  per_k_fits
}
```
