# Safety floor via three-way mean-gate — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Extend pigauto's post-training calibration from a 1-D `(1-r)·BM + r·GNN` blend to a 3-way simplex `r_BM·BM + r_GNN·GNN + r_MEAN·MEAN` so that pigauto is never worse than the grand-mean baseline on validation, by construction, with ≤ 2 % sampling-noise slack verified on five taxa.

**Architecture:** Add a `safety_floor = TRUE` default argument to `impute()` / `fit_pigauto()`. Extend the existing `calibrate_gates()` helper (which already does half-A/half-B cross-validation) to search a 231-point simplex grid when `safety_floor = TRUE`, returning three per-column weights instead of one. Store `mean_baseline_per_col` on the `pigauto_fit`. Update the predict blend to `r_BM·BM + r_GNN·GNN + r_MEAN·MEAN` with `%||%` fallback for legacy v0.9.1 fits (which lack the mean term). No changes to the training loop, torch model, or conformal machinery.

**Tech Stack:** R ≥ 4.0, `testthat` 3rd edition, `roxygen2`, `devtools`, `torch` ≥ 0.11, `Rphylopars` (Suggests), `ape`, `stats` (base).

**Working branch:** `feature/safety-floor-mean-gate` (already created, spec commit `ec23fd5` landed).

# R/calibration_three_way.R
# Safety-floor (three-way mean-gate) calibration helpers.
# See specs/2026-04-23-safety-floor-mean-gate-design.md.

# Build the 3-simplex grid at resolution `step`.
#
# Returns an integer-divided grid of (r_bm, r_gnn, r_mean) points where
# each row sums to exactly 1 and each entry is a non-negative multiple
# of `step`. At step = 0.05 this is 231 rows; step = 0.25 is 15 rows.
#
# The (1, 0, 0), (0, 1, 0), and (0, 0, 1) corners are always included:
# (0, 0, 1) is the grand-mean baseline candidate that gives the safety
# invariant `calibrated_loss <= mean_loss` by construction.
#
# @param step numeric in (0, 1]. Default 0.05 (231 points).
# @return numeric matrix with columns c("r_bm", "r_gnn", "r_mean").
#
# @noRd
simplex_grid <- function(step = 0.05) {
  stopifnot(is.numeric(step), length(step) == 1L, step > 0, step <= 1)
  k   <- as.integer(round(1 / step))
  if (abs(k * step - 1) > 1e-8) {
    stop("step must evenly divide 1; got step = ", step)
  }
  n_rows <- (k + 1L) * (k + 2L) / 2L
  out <- matrix(NA_real_, nrow = n_rows, ncol = 3L)
  row <- 1L
  for (i in 0:k) {
    for (j in 0:(k - i)) {
      out[row, ] <- c(i, j, k - i - j) / k
      row <- row + 1L
    }
  }
  colnames(out) <- c("r_bm", "r_gnn", "r_mean")
  out
}
```

# Compute the safety-floor mean-baseline scalar for one latent column.
#
# Returns a single numeric on the same latent scale as the column's
# bm_val / gnn_val (z-score for continuous/count/ordinal/proportion,
# logit for binary/categorical, z-score of magnitude for zi_count).
#
# For continuous-family columns: plain training grand mean on the
# already-z-scored `x_col`. This is approximately 0 (not exactly 0
# because z-scoring uses ALL preprocessed observations including ones
# that later become held-out test / val).
#
# For binary / categorical sub-columns: class-1 frequency clipped to
# [0.01, 0.99], then `qlogis()` to move to logit scale.
#
# @param x_col       numeric, one latent column, length n_obs
# @param train_mask  logical, TRUE for training-observed rows
# @param trait_type  character, one of "continuous", "count", "ordinal",
#                    "proportion", "zi_mag", "binary", "categorical"
# @return scalar numeric
#
# @noRd
mean_baseline_scalar <- function(x_col, train_mask, trait_type) {
  stopifnot(length(x_col) == length(train_mask),
            is.logical(train_mask),
            length(trait_type) == 1L)
  keep <- train_mask & !is.na(x_col)
  if (sum(keep) == 0L) return(0)
  if (trait_type %in% c("continuous", "count", "ordinal",
                          "proportion", "zi_mag")) {
    mean(x_col[keep])
  } else if (trait_type %in% c("binary", "categorical")) {
    p <- mean(x_col[keep])
    p <- pmin(pmax(p, 0.01), 0.99)
    qlogis(p)
  } else {
    stop("mean_baseline_scalar: unsupported trait_type = ", trait_type)
  }
}
```

resolve_one_split <- function(ha, hb) {
  min_abs_a <- if (is_discrete) 2 / max(length(ha), 1L) else 0
  min_abs_b <- if (is_discrete) 1 / max(length(hb), 1L) else 0

  # Candidate grid is either 1-D (legacy) or simplex (safety_floor).
  cand_grid <- if (safety_floor) simplex else {
    # Legacy 1-D: represent each g as the triple (1-g, g, 0) for code unity.
    matrix(c(1 - gate_grid, gate_grid, rep(0, length(gate_grid))),
           ncol = 3L,
           dimnames = list(NULL, c("r_bm", "r_gnn", "r_mean")))
  }

  # Baseline loss at the "pure BM" point (1, 0, 0) on half_a for the
  # relative-gain floor check.
  loss_a_pure_bm <- cal_mean_loss(c(1, 0, 0), ha)
  best_w         <- c(1, 0, 0)
  best_la        <- loss_a_pure_bm
  for (ci in seq_len(nrow(cand_grid))) {
    w_try <- cand_grid[ci, ]
    la    <- cal_mean_loss(w_try, ha)
    if (is.finite(la) && la < best_la) {
      best_la <- la
      best_w  <- w_try
    }
  }

  # Half-B verification: require the winning w's half-B loss to also
  # beat the pure-BM half-B loss by `cal_min_rel_gain` AND by the
  # discrete absolute cell floor.
  loss_b_pure_bm <- cal_mean_loss(c(1, 0, 0), hb)
  loss_b_best    <- cal_mean_loss(best_w,    hb)
  rel_gain_b     <- (loss_b_pure_bm - loss_b_best) / max(loss_b_pure_bm, 1e-12)
  abs_gain_b     <- loss_b_pure_bm - loss_b_best
  if (rel_gain_b < cal_min_rel_gain ||
      (is_discrete && abs_gain_b < min_abs_b)) {
    best_w <- c(1, 0, 0)   # reject; fall through to pure BM
  }

  best_w   # length-3 vector
}

# script/regress.R
# Full-canary regression check for safety_floor = TRUE.
# Runs pigauto on 5 taxa at n = 1000 each with safety_floor = TRUE vs
# FALSE, enforces:
#   - continuous RMSE_on <= 1.02 * RMSE_off on each vertebrate taxon
#   - discrete acc_on   >= acc_off - 0.01 on each discrete trait
#   - plants continuous RMSE_on <= 1.02 * mean_baseline_RMSE
#   - conformal coverage_on in [0.90, 1.00] on every trait
# Writes script/regress_result.json; exits non-zero on any failure.
#
# Invocation:
#   Rscript script/regress.R
#
# Wall ~20 min on Apple Silicon MPS.

suppressPackageStartupMessages({
  library(pigauto)
  library(jsonlite)
})
set.seed(2026L)

N_PER_TAXON  <- 1000L
SEED         <- 2026L
MISS_FRAC    <- 0.30
N_IMPS       <- 10L
EPOCHS       <- 300L
TOL_RMSE     <- 1.02
TOL_ACC      <- 0.01
TOL_COV_LO   <- 0.90
TOL_COV_HI   <- 1.00

cat_line <- function(...) cat(format(Sys.time(), "[%H:%M:%S] "), ..., "\n", sep = "")
taxa_results <- list()

bench_one <- function(taxon, df_truth, tree, cont_cols, disc_cols) {
  cat_line(sprintf("== %s: n = %d, p_cont = %d, p_disc = %d ==",
                    taxon, nrow(df_truth), length(cont_cols), length(disc_cols)))
  # 30% MCAR mask on each test column
  df <- df_truth
  mask <- matrix(FALSE, nrow = nrow(df),
                 ncol = length(c(cont_cols, disc_cols)),
                 dimnames = list(NULL, c(cont_cols, disc_cols)))
  for (v in c(cont_cols, disc_cols)) {
    ok <- which(!is.na(df_truth[[v]]))
    idx <- sample(ok, size = round(MISS_FRAC * length(ok)))
    mask[idx, v] <- TRUE
    df[[v]][idx] <- if (v %in% cont_cols) NA_real_ else NA
  }
  # Fit off and on
  t0 <- proc.time()[["elapsed"]]
  res_off <- pigauto::impute(df, tree, safety_floor = FALSE,
                               epochs = EPOCHS, n_imputations = N_IMPS,
                               verbose = FALSE, seed = SEED)
  wall_off <- proc.time()[["elapsed"]] - t0
  t0 <- proc.time()[["elapsed"]]
  res_on  <- pigauto::impute(df, tree, safety_floor = TRUE,
                               epochs = EPOCHS, n_imputations = N_IMPS,
                               verbose = FALSE, seed = SEED)
  wall_on  <- proc.time()[["elapsed"]] - t0
  # Metrics per trait
  out <- list(taxon = taxon, n = nrow(df_truth),
              wall_off_s = wall_off, wall_on_s = wall_on,
              traits = list(), pass = TRUE)
  for (v in cont_cols) {
    if (!any(mask[, v])) next
    truth_v <- df_truth[[v]][mask[, v]]
    rmse_off <- sqrt(mean((res_off$completed[[v]][mask[, v]] - truth_v)^2,
                          na.rm = TRUE))
    rmse_on  <- sqrt(mean((res_on$completed[[v]][mask[, v]]  - truth_v)^2,
                          na.rm = TRUE))
    rmse_mean <- sqrt(mean((mean(df[[v]], na.rm = TRUE) - truth_v)^2,
                           na.rm = TRUE))
    cov_on <- if (!is.null(res_on$prediction$conformal_lower)) {
      # conformal coverage on this column
      lo <- res_on$prediction$conformal_lower[, v][mask[, v]]
      hi <- res_on$prediction$conformal_upper[, v][mask[, v]]
      mean(truth_v >= lo & truth_v <= hi, na.rm = TRUE)
    } else NA_real_
    pass_rmse  <- rmse_on <= rmse_off * TOL_RMSE
    pass_mean  <- rmse_on <= rmse_mean * TOL_RMSE
    pass_cov   <- is.na(cov_on) || (cov_on >= TOL_COV_LO & cov_on <= TOL_COV_HI)
    pass_all   <- pass_rmse && pass_mean && pass_cov
    out$traits[[v]] <- list(
      type = "continuous",
      rmse_off = rmse_off, rmse_on = rmse_on, rmse_mean = rmse_mean,
      coverage_on = cov_on,
      pass_rmse = pass_rmse, pass_mean = pass_mean, pass_cov = pass_cov,
      pass = pass_all)
    if (!pass_all) out$pass <- FALSE
  }
  for (v in disc_cols) {
    if (!any(mask[, v])) next
    truth_v <- df_truth[[v]][mask[, v]]
    acc_off <- mean(res_off$completed[[v]][mask[, v]] == truth_v, na.rm = TRUE)
    acc_on  <- mean(res_on$completed[[v]][mask[, v]]  == truth_v, na.rm = TRUE)
    pass_acc <- acc_on >= acc_off - TOL_ACC
    out$traits[[v]] <- list(
      type = "discrete",
      acc_off = acc_off, acc_on = acc_on,
      pass = pass_acc)
    if (!pass_acc) out$pass <- FALSE
  }
  out
}
